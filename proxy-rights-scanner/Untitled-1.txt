// types/index.ts
export interface ProxyRight {
  id: string;
  category: string;     // 예: "금융기관 거래", "부동산 관리" 등
  description: string;  // 대리권의 상세 범위 및 내용
}

export interface DocumentData {
  documentId: string;
  imageUrl: string;
  proxyRights: ProxyRight[];
  analyzedAt: string; // ISO 8601 날짜 문자열
}
